# generate data

with open('../data/20170405/edgelist_userfans.txt','w') as fout:
    fin = open('../data/20170405/userfans.txt')
    fin.readline()
    fout.write('source\ttarget\n')
    for line in fin:
        fields = line.strip().split()
        uid_to = fields[0].replace('"','')
        uid_from = fields[1].replace('"','')
        fout.write('%s\t%s\n'%(uid_from,uid_to))
